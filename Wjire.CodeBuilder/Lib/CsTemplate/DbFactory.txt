namespace {NameSpaceName}.DbContext
{

    /// <summary>
    /// 数据库 {DbName} 连接工厂
    /// </summary>
    public static partial class {DbName}DbFactory
    {
       
        private static readonly string {DbName}Read = "{DbName}Read";
        private static readonly string {DbName}Write = "{DbName}Write";

        /// <summary>
        /// 创建事务
        /// </summary>
        /// <returns></returns>
        public static IUnitOfWork CreateTransaction()
        {
            return new UnitOfWork({DbName}Write);
        }

        /// <summary>
        /// 创建单连接
        /// </summary>
        /// <returns></returns>
        public static IUnitOfWork CreateSingle()
        {
            return new SingleConnection({DbName}Read);
        }

    }
}
